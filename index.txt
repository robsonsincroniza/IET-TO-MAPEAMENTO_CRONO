        }

        // Exportação para Excel (CSV compatível)
        function exportTableToCSV(filename) {
            let csv = [];
            let rows = document.querySelectorAll("#formacao-table tr");
            
            for (let i = 0; i < rows.length; i++) {
                let row = [], cols = rows[i].querySelectorAll("td, th");
                
                // Ignora a última coluna de "Ação" na exportação
                for (let j = 0; j < cols.length - 1; j++) {
                    let val = cols[j].innerText;
                    
                    // Se for célula com input, textarea ou select, pega o value
                    if(cols[j].querySelector('input')) val = cols[j].querySelector('input').value;
                    if(cols[j].querySelector('textarea')) val = cols[j].querySelector('textarea').value;
                    if(cols[j].querySelector('select')) val = cols[j].querySelector('select').value;
                    
                    // Limpa quebras de linha para não corromper o CSV e encapsula em aspas
                    row.push('"' + val.replace(/"/g, '""').replace(/\n/g, ' - ') + '"');
                }
                csv.push(row.join(";"));
            }
            
            let csvFile = new Blob(["\uFEFF"+csv.join("\n")], {type: "text/csv;charset=utf-8;"});
            let downloadLink = document.createElement("a");
            downloadLink.download = filename;
            downloadLink.href = window.URL.createObjectURL(csvFile);
            downloadLink.style.display = "none";
            document.body.appendChild(downloadLink);
            downloadLink.click();
            document.body.removeChild(downloadLink);
        }
    </script>
</body>
